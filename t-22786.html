Nothing special! - Passage returns zero if successful.


long ret
long info(PSMAXSIZE)
string progname(512)
long procid

function main()
{
ret = shell("${BSE}/scripts/passage",SHELL_MWINDOW)
if ret > 0 then
procid = -1
procid = pstat( procid, progname, info)
while procid > 0
procid = pstat(procid, progname, info)
kill(procid)
endwhile
endif
}

long info(PSMAXSIZE)

{
