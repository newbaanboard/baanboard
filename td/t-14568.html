select tcmcs032.*
from tcmcs032
where tcmcs032._index1 = {:ccty,:cvat}
and tcmcs032.edat >= :efdt
order by tcmcs032.edat desc
as set with 1 rows
selectdo
|tcmcs032.pvat is the tax rate
selectemty
|no tax rate found
endselect

|ccty ,cvat and efdt may be variables in your report script.
