hi,
you can refer this example

choice.cont.process:
on.choice:
execute(print.data)

choice.print.data:
on.choice:
if rprt_open() then
read.main.table()
rprt_close()
else
choice.again()
endif


|****************************** field section *********************************

field.orno.f:
when.field.changes:
orno.t = orno.f


|****************************** function section ******************************

functions:

function read.main.table()
{
select tdpur400.*
from tdpur400
where tdpur400._index1 inrange {:orno.f}
and {:orno.t}
order by tdpur400._index1
selectdo
rprt_send()
endselect
}
