select *
from whinr110
where whinr110.kost=2 and
not exists (select whinr110.item
from whinr110
wehre whinr110.kost<>2
and whinr110.item= :whinr110.item)[/INDENT][/INDENT]
selectdo
endselect

OR


select *
from whinr110
where whinr110.kost=2
and whinr110.item not in (select whinr110.item
from whinr110
wehre whinr110.kost<>2)
selectdo
endselect
