string sh.cmd(200)

function create.shell.command.for.mail()
{
sh.cmd = "mailx "
&"-s "
&strip$(subject)
&" -r "
&strip$(receipient)
&" "
&strip$(sender)
&" < "
&strip$(path.file)

}

ret.mail = shell(sh.cmd, 64)
