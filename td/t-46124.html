you can use following function to open excel or you can write your own program by using app_start function. Don't forget to include following tool Dll

#pragma used dll "ottdllbw"

run.app("excel.exe C:\template.xls")

function long run.app(domain dtmail.dsca prog)
{
long app_id,event(EVTMAXSIZE)
domain tcbool terminated

app_id = app_start(prog, "", "", "", "")
terminated = false
suspend(1000)
if app_status(app_id) then
while not terminated and next.event(event)
on case evt.type(event)
case EVTTERMINATION:
if evt.termination.object(event) = app_id then
terminated = true
endif
break
endcase
endwhile
endif
return(1)
}
