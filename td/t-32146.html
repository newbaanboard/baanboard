function read.itemtext()
{

item.text.file = ""

string itext(200)
string ibuf(80,40)
long i

itext = ""

for i = 1 to
text.to.buf("tiitm001.txta", "language", 1, ibuf)
itext = strip$(itext)&" "&shiftl$(shiftr$((ibuf(1,i)))
endfor

item.text.file = itext
}
