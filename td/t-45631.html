Script:

declaration:

table ttablename
long ret ,fcmp

before.program:

fcmp = your_target_company_number

ret = db.drop.table(ttablename,DB.IGNORE.ALL.REFS, fcmp)
if ret then
if db.clear.table(ttablename, no.rollback, fcmp) then
execute(abort.program)
endif
ret = 0
else
db.create.table(ttablename, fcmp)
endif
