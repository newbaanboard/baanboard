Hi,

try this:


field.2:
before.input:
on case field.1:
case x: attr.zoomcode = Z.SESSION
attr.zoomsession$ = "ses1m000"
attr.zoomreturn$ = "retfield1"
break
case y: attr.zoomcode = Z.SESSION
attr.zoomsession$ = "ses2m000"
attr.zoomreturn$ = "retfield2"
break
case z: attr.zoomcode = Z.SESSION
attr.zoomsession$ = "ses3m000"
attr.zoomreturn$ = "retfield3"
break
default: attr.zoomcode = 0
endcase
