function read.purchase.item.text()
{

ektext = ""

string ptext(1024)

for p = 1 to
text.to.buf("tiitm001.txtp", "3", 10, pbuf)
ptext = strip$(ptext)&" "&strip$(pbuf(1,p))
endfor

ektext = ptext

}
