example :

before.field.1:
before.layout:
footer.print = false
count = 0

detail.1:
before.layout:
if count => 10 then
footer.print = true
count = 0
page()
endif

after.layout:
count = count + 1

footer.1:
before.layout:
if not footer.print then
lattr.print = false
endif

after.field.1:
before.layout:
if footer.print then
lattr.print = false
endif
