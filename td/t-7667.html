Found this:


select tfgld410.*
from tfgld410 for update
where tfgld410._index1 =
{ :tfgld418.ocom,
:tfgld418.tror,
:tfgld418.fitr,
:tfgld418.trdt,
:tfgld418.trtm,
:tfgld418.sern,
0 }
as set with 1 rows
selectdo |#93247-
1.n
endselect
select tfgld417.*
from tfgld417 for update
where tfgld417._index1 =
{ :tfgld418.ocom,
:tfgld418.tror,
:tfgld418.fitr,
:tfgld418.trdt,
:tfgld418.trtm,
:tfgld418.sern}
as set with 1 rows
selectdo |#93247-
1.n
endselect


Hope it helps.
