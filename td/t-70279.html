Hi you can use this piece of code..

validate.textline()


function void validate.textline()
{
if isspace(lattr.prline) then
lattr.print = false
else
if lattr.prline(1;1) = ">" then
lattr.prline = lattr.prline(2)
else
if lattr.prline(1;1) = "<" then
lattr.print = false
endif
endif
endif
}
