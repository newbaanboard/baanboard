function long itfgld0032.num.to.time(domain tctime trans.time)
{
domain tcmcs.byte hh
domain tcmcs.byte mm
domain tcmcs.byte ss

hh = trans.time / 3600
mm = (trans.time - (hh * 3600)) / 60
ss = trans.time - (hh * 3600) - (mm * 60)
return((hh * 10000) + (mm * 100) + ss)
}
