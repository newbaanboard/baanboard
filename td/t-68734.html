If you are using Oracle db, you can use the following query (use the appropriate schema/db-owner name) -

select t$cpac, t$cmod, t$cfrm, t$fldn
from ERPLN.tttadv301000
where t$cpac = 'tc' and t$cust = 'cust'
minus
select t$cpac, t$cmod, t$cfrm, t$fldn
from ERPLN.tttadv301000
where t$cpac = 'tc' and t$cust = 'stnd'
