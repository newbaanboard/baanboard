public string GetTableDesc(string table)
{
string tabledesc = "";
try
{
string sql = "select t_desc from tttadv130000 where t_cpac='" + table.Substring(1, 2) + "' and t_rkey='" + table.Substring(3, 6) + "'";
IDataReader reader = _database.GetReaderForSql(sql);
reader.Read();
tabledesc = (string)reader["t_desc"];
reader.Close();
}
finally
{
}
return tabledesc;
}

public string GetColumnDesc(string table,string column)
{
string columndesc = "";
string sql="";
try
{
sql = "select top 1 t_clab from tttadv422000 where t_cpac='" + table.Substring(1, 2) + "' and t_cmod='" + table.Substring(3, 3) + "' and t_flno='" + table.Substring(6, 3) + "' and t_fdnm='" + column.Substring(2, 4) +"'";
IDataReader reader = _database.GetReaderForSql(sql);
while(reader.Read())
columndesc = (string)reader["t_clab"];
reader.Close();
if (columndesc.Trim() == "")
sql = "select top 1 t_desc from tttadv140000 where t_cpac='" + table.Substring(1, 2) + "' and t_clab='" + table.Substring(1,8) + "." + column.Substring(2, 4) + "'";
else
sql ="select top 1 t_desc from tttadv140000 where t_clab='" + columndesc + "'";
reader = _database.GetReaderForSql(sql);
while(reader.Read())
columndesc = (string)reader["t_desc"];
reader.Close();
}
finally
{
}
return columndesc;
}
