Simple alternative:


function fill.report()
{
long first.rec

first.rec = true

select ...
where ...
selectdo
if first.rec then
| do whatever you need to do with the first record
first.rec = false
endif
| Process normally
rprt_send()
selecteos
| Last record processing
rprt_send()
endselect
}
