field.tcibd001.item:
before.zoom:
if (condition = true) then
attr.zoomsession$ = "session1"
attr.zoomreturn$ = "fieldX"
else
attr.zoomsession$ = "session2"
attr.zoomreturn$ = "fieldX"
endif
