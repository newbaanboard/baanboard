you have to test the following constants:

OS_WINDOWS_NT
OS_WINDOWS_95
OS_UNIX
OS_OS400



ret = ostype()
on case ret
case OS_WINDOWS_NT:
str = "WINDOWS/NT"
break
case OS_WINDOWS_95:
str = "WINDOWS 95"
break
case OS_UNIX:
str = "UNIX"
break
case OS_OS400:
str = "AS 400"
break
endcase
message(str)
