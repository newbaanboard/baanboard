long conv.time.tctmhs
long hh
long mm
long conv.time.tctime
string hhmm(5)

function decode.time.tctime.tctmhs(domain tctime iptm)
{
conv.time.tctmhs = iptm
hh = (conv.time.tctmhs / 3600)
mm = ((conv.time.tctmhs / 60) - (hh * 60))

if (mm < 10) then
conv.time.tctmhs = lval(str$(hh) & "0" & str$(mm))
else
conv.time.tctmhs = lval(str$(hh) & str$(mm))
endif
}


function decode.time.tctmhs.tctime(domain tctmhs iptm)
{
hhmm = (str$(iptm + 10000))
conv.time.tctime = (((lval(hhmm(2;2))) * 3600) + ((lval(hhmm(4;2))) * 60))
}

Hi maurixgr,

To convert date into string

extern string date.string(8)
extern domain tcpono year
extern domain tcsrnb month, days


date.string = date.to.inputstr$(tdpur045.date,"%D002,3",8)
year = val(date.string(5;4))
month = val(date.string(3;2))
days = val(date.string(1;2))


the above command convert date of tdpur045 into string & store it in variable date.string

After that, the string split into days, month, year
