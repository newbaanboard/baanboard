Use seq.puts below :

fp = seq.open(spool.fileout, "a+")

item = "0123456789"
Quantity = "12"
Supplier = "680014"

ligne = "'"&Item&"'"&","&Quantity&","&"'"&Supplier&"'"

ret = seq.puts(ligne, fp)
ret = seq.close(fp)
