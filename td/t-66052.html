select cisli245.pono,
cisli245.ddat,
cisli245.ityp,
cisli245.idoc,
cisli245.slso,
cisli245.amti,
cisli245.ldai,
cisli245.txah
from cisli245
where cisli245.ityp = :cisli205.ityp and
cisli245.idoc = :cisli205.idoc
order by cisli245.ityp,
