Found the solution.....

# B (24.02.2012)
# Script to write BaaN Licmon to G:\temp\licuse.txt

BSE=G:\Baan
BSE_TMP=G:\Baan\tmp
Temp=G:\temp

export BSE BSE_TMP Temp

# Write used BaaN Licence to File
G:\Baan\bin\licmon -u >> G:\temp\licuse.txt
G:\Baan\bin\licmon -w >> G:\temp\licuse.txt
