You can also use...

select t.mser
from table
where pdno = x
as set with 1 rows
selectdo
| It will give u first serial no for that production
endselect



select t.mser
from table
where pdno = x
order by t.mser desc
as set with 1 rows
selectdo
| It will give u last serial no for that production
endselect
