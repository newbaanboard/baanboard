Hello,

use something like:

if tc.is.html.ui() = FALSE then
|Transfer to client -> Excel
fehler = server2client("Testerdatenexport_" & logname$ & ".csv",
"C:\temp\" & trim$("Testerdatenexport_" & logname$ & ".csv"), FALSE, TRUE)
|Start Excel
start.application.local("excel.exe " & "C:\temp\" & trim$("Testerdatenexport_" & logname$ & ".csv"), TRUE, fehler)
else
fehler = client.show.file("Testerdatenexport_" & logname$ & ".csv", TRUE, "Testerdatenexport_" & logname$ & ".csv")
endif |html ui
endif |to XLX
