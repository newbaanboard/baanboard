acutally it is permission problem , i had the similar problem , here is the remedy:
1 : DELETE DEFAULT BAAN LOGIN baan AND GROUP baan4dbg

2: EXECUTE BELOW QUERY

use master
EXEC sp_addlogin baan_id,password,baan_dbname
use baan4db -- or your baan dbname here
EXEC sp_changedbowner baan4db, baan
use baan4db
EXEC sp_addrole baan4dbg
use baan4db
grant create table to baan4dbg
use baan4db
EXEC sp_adduser baan
EXEC sp_addrolemember baan4dbg, baan

3. AFTER THIS SET PERMISSION of user baan TO DB_OWNER TO DATABASE BAAN4DB
