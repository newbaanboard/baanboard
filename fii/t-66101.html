Referring to the below function in report script, result of tfgld.desc.amt will show the value as **999999999.99**
Example **1280095.20**. Is there any way to produce result in comma format as **1,280,095.20**.


function show.words()
{
string decimal.amount.str(2)
decimal.amount.str = edit$(((tfcmg103.amnt * 100)\ 100), "99")
pence = decimal.amount.str

string amount(10)
long decpos
long length
long i
amount = str$(tfcmg103.amnt)
decpos = rpos(amount,".")
if decpos <> 0 then
amount = amount(1;decpos-1)
endif
length = len(amount)

tfgld.desc.amt= "**" & strip$(shiftr$(amount)) & "."&pence& "**"
}
