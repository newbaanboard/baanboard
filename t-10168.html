sub BaanSession

BaanSess = "ttaad5203m000"
KeyField = "ttaad500.cjob"
KeyValue = "TEST"

Set BaanObj = CreateObject("Baan4.Application")

BaanObj.ParseExecFunction "ottstpapihand", "stpapi.find(" & Chr(34) & BaanSess & Chr(34) & ")"

BaanObj.ParseExecFunction "ottstpapihand", "stpapi.put.field(" & Chr(34) & BaanSess & Chr(34) & "," & Chr(34) & KeyField & Chr(34) & "," & Chr(34) & KeyValue & Chr(34) & ")"

temp = "stpapi.continue.process (" & Chr(34) & BaanSess & Chr(34) & "," & Chr(34) & "free$" & Chr(34) & ")"
BaanObj.ParseExecFunction "ottstpapihand", temp

BaanObj.ParseExecFunction "ottstpapihand", "stpapi.end.session(" & Chr(34) & BaanSess & Chr(34) & ")"

BaanObj.Quit

Set BaanObj = Nothing

end sub
