stpapi.put.field("ticst0101m100", "ticst001.pdno",i.pdno)
stpapi.put.field("ticst0101m100", "ticst001.pono",str$(i.pono))
stpapi.find("ticst0101m100", err.msg)
ret = stpapi.mark("ticst0101m100", err.msg)
ret = stpapi.synchronize.dialog("ticst0101m100","modify",err.msg)
if ret then
stpapi.handle.subproc("ticst0101m100","timfc0101m000","search")
stpapi.form.command( "ticst0101m100", 2, "timfc0101m000", ret.err)
stpapi.form.command("timfc0101m000",5,"initiate.order", ret.err)
stpapi.end.session("timfc0101m000",err.msg)
stpapi.end.session("ticst0101m100",err.msg)
if isspace(ret.err) then
return(true)
endif
endif
stpapi.end.session("ticst0101m100",err.msg)
return(false)
